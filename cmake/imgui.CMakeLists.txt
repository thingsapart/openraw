# This file is copied into the imgui source directory by FetchContent
# to provide CMake build system integration for the library.

cmake_minimum_required(VERSION 3.16)
project(imgui-wrapper)

# Define the library target with its source files.
# Paths are relative to the imgui source root.
add_library(imgui STATIC
    imgui.cpp
    imgui_draw.cpp
    imgui_tables.cpp
    imgui_widgets.cpp
    imgui_demo.cpp
    backends/imgui_impl_sdl2.cpp
    backends/imgui_impl_opengl3.cpp
)

# Expose the necessary include directories publicly.
# Any target that links against 'imgui' will automatically inherit these.
target_include_directories(imgui PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/backends
)
